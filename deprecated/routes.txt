from fastapi import FastAPI, Request, Response, Cookie
from fastapi.templating import Jinja2Templates
from fastapi.responses import HTMLResponse, RedirectResponse


import json

app = FastAPI(title="SOFA Frontend Refactor")

templates = Jinja2Templates(directory="app/templates")

# ---------------------------------------
# Hilfsfunktion zum Auslesen des Users
# ---------------------------------------
def get_current_user(sofa_user: str | None):
    if sofa_user:
        return json.loads(sofa_user)
    return None

# ---------------------------------------
# Dashboard (offen)
# ---------------------------------------
@app.get("/", response_class=HTMLResponse)
def index(request: Request, sofa_user: str | None = Cookie(default=None)):
    user = get_current_user(sofa_user)
    return templates.TemplateResponse("dashboard.html", {"request": request, "user": user})


# ---------------------------------------
# Login Overlay / POST (extern)
# ---------------------------------------
@app.get("/login", response_class=HTMLResponse)
def login_page(request: Request):
    # Einfach Overlay-Form oder Login-Template
    return templates.TemplateResponse("login.html", {"request": request})

@app.post("/login")
def login(request: Request, response: Response):
    # Platzhalter: hier später externes Backend aufrufen
    # Beispiel: user_data = call_auth_api(username, password)
    user_data = {
        "username": "demo",
        "first_name": "Max",
        "roles": ["Admin"]
    }

    # Cookie setzen
    cookie_value = json.dumps(user_data)
    response = RedirectResponse(url="/", status_code=303)
    response.set_cookie(
        key="sofa_user",
        value=cookie_value,
        httponly=True,
        max_age=3600*8
    )
    return response

@app.get("/logout")
def logout(response: Response):
    response = RedirectResponse(url="/", status_code=303)
    response.delete_cookie("sofa_user")
    return response
